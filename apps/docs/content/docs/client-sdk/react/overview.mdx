The `@syncular/client-react` package provides React hooks and components for Syncular.

### What you'll learn

- How to install and configure the React SDK
- Setting up `SyncProvider` with transport and subscriptions
- Querying synced data reactively with `useSyncQuery`
- Making mutations from React components
- All available hooks at a glance

## Installation

```bash
npm install @syncular/client @syncular/client-react @syncular/transport-http
```

## Quick Start

```tsx
import { createSyncularReact } from '@syncular/client-react';
import { defineClientSync } from '@syncular/client';
import type { SyncClientDb } from '@syncular/client';

type Task = { id: string; title: string; completed: number; created_at: string };
type Db = SyncClientDb & { tasks: Task };

const { SyncProvider, useSyncQuery, useMutations } = createSyncularReact<Db>();
const sync = defineClientSync<Db, ['user:{user_id}'], { actorId: string }>({});

sync.addHandler({
  table: 'tasks',
  scopes: ['user:{user_id}'],
  subscribe: ({ identity }) => ({
    id: `my-tasks-${identity.actorId}`,
    table: 'tasks',
    scopes: { user_id: identity.actorId },
  }),
});

function App() {
  return (
    <SyncProvider
      db={db}
      transport={transport}
      sync={sync}
      identity={{ actorId: userId }}
      clientId={deviceId}
    >
      <TaskList />
    </SyncProvider>
  );
}

function TaskList() {
  const { data: tasks, isLoading } = useSyncQuery(({ selectFrom }) =>
    selectFrom('tasks').selectAll().orderBy('created_at', 'desc')
  );
  const m = useMutations();

  if (isLoading) return <div>Loading...</div>;

  return (
    <ul>
      {tasks?.map((task) => (
        <li key={task.id} onClick={() => m.tasks.update(task.id, { completed: 1 })}>
          {task.title}
        </li>
      ))}
    </ul>
  );
}
```

## Available Hooks

| Hook | Description |
|------|-------------|
| `useSyncQuery` | Query local data with reactive updates |
| `useMutations` | Make changes that sync |
| `useMutation` | Single-table mutations |
| `useSyncStatus` | Monitor sync status |
| `useSyncEngine` | Access sync engine directly |
| `useSyncConnection` | Monitor WebSocket connection |
| `useTransportHealth` | Monitor realtime/polling fallback health |
| `useSyncProgress` | First-class bootstrap/sync progress |
| `useSyncSubscriptions` | Read persisted subscription metadata |
| `useSyncSubscription` | Read one persisted subscription metadata row |
| `useConflicts` | Get pending conflicts |
| `useNewConflicts` | Track newly surfaced unresolved conflicts for toasts/banners |
| `useResolveConflict` | Resolve conflicts |
| `useOutbox` | Get outbox status |
| `useSyncContext` | Access raw sync context |
| `useSyncGroup` | Aggregate multiple sync engines/channels |
