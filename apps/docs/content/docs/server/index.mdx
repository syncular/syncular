## Installation

```bash
npm install @syncular/server @syncular/server-hono
npm install @syncular/server-dialect-postgres kysely pg hono
```

## Quick Setup

```typescript
import { Hono } from 'hono';
import { createSyncRoutes } from '@syncular/server-hono';
import { ensureSyncSchema } from '@syncular/server';
import { createPostgresServerDialect } from '@syncular/server-dialect-postgres';

const app = new Hono();

// 1) Create a sync dialect and ensure the sync tables exist
const dialect = createPostgresServerDialect();
await ensureSyncSchema(db, dialect);

// 2) Register your server table handlers
const handlers = [tasksHandler];

// 3) Mount the sync routes
const syncRoutes = createSyncRoutes({
  db,
  dialect,
  handlers,
  authenticate: async (ctx) => ({ actorId: 'user-123' }),
});

app.route('/sync', syncRoutes);

export default app;
```
