# Quick Start

Pass your generated `AppSchema` into the testkit. The fixture opens a real
SQLite replica, installs app and Syncular runtime migrations, and uses the same
generated mutations as production code.

```rust
use syncular_testkit::{
    assert_outbox_empty, assert_table_has_row, AppFixtureOptions,
    AppTestServer, open_app_client_with_server,
};

use my_app::generated::{diesel_tables::TaskRow, schema, syncular};
use syncular::prelude::SyncularGeneratedMutationsExt;

#[test]
fn two_clients_converge() {
    let app_schema = my_app::generated::app_schema();
    let server = AppTestServer::new(app_schema);

    let mut writer = open_app_client_with_server(
        app_schema,
        server.clone(),
        AppFixtureOptions {
            client_id: "writer".to_string(),
            ..AppFixtureOptions::default()
        },
    ).unwrap();

    let mut reader = open_app_client_with_server(
        app_schema,
        server,
        AppFixtureOptions {
            client_id: "reader".to_string(),
            ..AppFixtureOptions::default()
        },
    ).unwrap();

    writer
        .client
        .mutations()
        .tasks()
        .insert(syncular::NewTask::new(
            "task-1",
            "From writer",
            "user-rust",
            Some("p0"),
        ))
        .unwrap();

    writer.client.sync_http().unwrap();
    assert_outbox_empty(&mut writer.client);

    reader.client.sync_http().unwrap();
    assert_table_has_row(&mut reader.client, "tasks", "id", "task-1");

    let rows: Vec<TaskRow> = reader
        .client
        .read(
            schema::tasks::dsl::tasks
                .select(TaskRow::as_select()),
        )
        .unwrap();

    assert_eq!(rows[0].title, "From writer");
}
```

## What this test proves

- generated mutations create local rows and outbox commits
- the stateful test server accepts the push
- another client can pull the row through the same app schema
- assertions inspect real SQLite state instead of mocks

Use this shape as the default before adding app-specific fixtures.
