# SDK Reference

## Installation

```bash
# Core SDK
npm install @syncular/client

# React integration
npm install @syncular/client-react
```

## Quick Example

```tsx
import { createSyncularReact } from '@syncular/client-react';
import { ClientTableRegistry, type SyncClientDb } from '@syncular/client';
import { createHttpTransport } from '@syncular/transport-http';

type Task = { id: string; title: string };
type Db = SyncClientDb & { tasks: Task };

const { SyncProvider, useSyncQuery, useMutations } = createSyncularReact<Db>();

function App() {
  const transport = createHttpTransport({ baseUrl: '/api' });
  const shapes = new ClientTableRegistry().register(tasksClientHandler);

  return (
    <SyncProvider
      db={db}
      transport={transport}
      shapes={shapes}
      actorId={userId}
      clientId={deviceId}
      subscriptions={[{ id: 'tasks', shape: 'tasks', scopes: {} }]}
    >
      <TaskList />
    </SyncProvider>
  );
}

function TaskList() {
  const { data: tasks } = useSyncQuery(({ selectFrom }) =>
    selectFrom('tasks').selectAll()
  );

  const m = useMutations();

  return (
    <ul>
      {tasks?.map(task => (
        <li key={task.id}>{task.title}</li>
      ))}
    </ul>
  );
}
```
